137 [[nodiscard]] G4ThreeVector apply_units(const double dflt_unit) const {
138 auto vec = G4ThreeVector(0, 0, 0);
139 auto unit = this->units.value_or(dflt_unit);
140 vec.setX(this->x * unit);
141 vec.setY(this->y * unit);
142 vec.setZ(this->z * unit);
143 return vec;
144 }
145 };
